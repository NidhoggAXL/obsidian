# 首相要知道是创造一个什么结构的数据栈：
```
如果用尾做栈顶，尾插尾删，要设计成双向链表，否则删除数据效率低(因为如果是不循环的链表找尾需要从头开始遍历，这样才可以找到尾，效率低)

如果用头做栈顶，头插头删，就可以设计成单链表
```

# 构建栈的结构体
顺序栈
```C
typedef int STDataType;
typedef struct Stack
{
	STDataType* a;//指向存储数据的指针
	int top;//栈顶的位置
	int capacity;//数据的容量
}ST;
```

# 接口的声明
```C
void StackInit(ST* ps);//初始化

void StackDestroy(ST* ps);//销毁
int StackEmpty(ST* ps);//判断栈是否为空(返回真和假)

void StackPush(ST* ps,STDataType x);//栈顶进行插入
void StackPop(ST* ps);//删除

STDataType StackTop(ST* ps);//取栈顶的数据
int StackSize(ST* ps);//栈里面有多少个数据

```

# 结构的定义
## 初始化
```C
//初始化
void StackInit(ST* ps)
{
	assert(ps);
	ps->a = NULL;
	//ps->top=0时是先放入数据在top++,top是指向栈顶数据的下一个
	//ps->top=-1时是先top++在存放数据，top指向栈顶数据
	ps->top = 0;
	ps->capacity = 0;
}
```

## 栈顶数据的插入
```C
//栈顶进行插入
void StackPush(ST* ps,STDataType x)
{
	assert(ps);
	if (ps->top == ps->capacity)
	{
		int newCapacity = ps->capacity == 0 ? 4 : ps->capacity * 2;
		STDataType* tmp = realloc(ps->a, sizeof(STDataType) * newCapacity);
		//这里的扩容也不和前面的相同，因为这里是结构体里面的内容进行扩容
		//所以不可以使用结构体指针，而是定义的数据类型指针
		assert(tmp);
		ps->a = tmp;
		ps->capacity = newCapacity;
	}
	ps->a[ps->top] = x;//把数据放到ps->top指向的位置
	ps->top++;//++指向后面一个数据
}
```

## 删除数据
```C
void StackPop(ST* ps)//删除数据
{
	assert(ps);
	assert(ps->top > 0);
	ps->top--;
}
```

## 取栈顶数据
返回的是定义得数据类型，不是结构体类型。
```c
STDataType StackTop(ST* ps)//取栈顶的数据
{
	assert(ps);
	assert(ps->top > 0);	
	return ps->a[ps->top - 1];
}
```

## 统计栈里面元素
```c
int StackSize(ST* ps)//栈里面有多少个数据
{
	assert(ps);
	return ps->top;
}
```

## 判断栈是否为空
```c

int StackSize(ST* ps)//栈里面有多少个数据
{
	assert(ps);
	return ps->top;
}
```

## 销毁数据
```c
void StackDestroy(ST* ps)//销毁
{
	assert(ps);
	free(ps->a);
	ps->a = NULL;
	ps->capacity = ps->top = 0;
}
```

# 总模块函数
## 主函数(text.c)
```C
#define _CRT_SECURE_NO_WARNINGS 1
#include "Stack.h"

void text1()
{
	ST st;
	StackInit(&st);
	StackPush(&st, 1);
	StackPush(&st, 2);
	StackPush(&st, 3);
	StackPush(&st, 4);

	StackPop(&st);//删除栈尾数据
	printf("%d \n", StackTop(&st));


	while (StackEmpty(&st))//判断栈是否为空)
	{
		printf("%d ", StackTop(&st));
		StackPop(&st);
	}
	StackDestroy(&st);
}
int main()
{
	text1();
	return 0;
}
```

		ps->capacity = newCapacity;

}

{
	assert(ps);
	if (ps->top > 0)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}
