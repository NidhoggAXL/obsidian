# 结构体的定义
```C
typedef int QDataType;

typedef struct QueueNode
{
	struct QueueNode* next;//递归的思路(单链表思路相同)
	QDataType data;
} QueueNode;

typedef struct Queue
{
	QueueNode* head;//指向头结构体的指针
	QueueNode* tail;//指向尾结构体的指针
}Queue;
```
这里使用两个结构体，是因为避免使用*二级指针*，这样定义方便一点。

# 函数的定义
## 初始化
```c
void QueueInit(Queue* pq)//初始化
{
	assert(pq);
	pq->head = NULL;
	pq->tail = NULL;
}
```

## 销毁
```C

void QueueDestroy(Queue* pq)//销毁
{
	assert(pq);
	QueueNode* cur = pq->head;
	while (cur != NULL)
	{
		QueueNode* next = cur->next;
		free(cur);
		cur = next;
	}
	pq->tail = pq->head = NULL;
}
```

## 插入数据
```C
void QueuePush(Queue* pq, QDataType x)//插入数据
{
	assert(pq);
	QueueNode* newnode = (QueueNode*)malloc(sizeof(QueueNode));
	newnode->data = x;
	newnode->next = NULL;
	if (QueueEmpty(pq))//判断结构体QueueNode数据是否为空
	{
		pq->head = pq->tail=newnode;
	}
	else
	{
		pq->tail->next = newnode;
		pq->tail = newnode;
	}
}
```

## bool判断队列是否为空
```c
bool QueueEmpty(Queue* pq)//判断队列是否为空
{
	assert(pq);
	return pq->head == NULL;//如果结果成立，则传的是bool类型的真(Ture),为假的化就是传flase
}

```


## 删除数据
```c
void QueuePop(Queue* pq)//删除数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//bool和assert的联合运用，当boo为真是因为有!则assert为假
	QueueNode* next = pq->head->next;
	free(pq->head);
	pq->head = next;
	if (QueueEmpty(pq))
	{
		pq->tail = NULL;
	}
}
```
这里为什么要对进行一下判断，对``free(pq->tail)``进行操作呢，因为当只有一个数据的时候``pq->head=pq->tail``，操作到``free(pq->head)``释放了``pq->head``指向的空间了，这是如果不对``pq->tail``进行处理的化，就会造成野指针，指向随机值，入下图phead都为空了，但是tail还指向一个地址。

## 取头和尾的数据
取数据时一定要注意是否数据为空的情况。
```C
QDataType QueueFront(Queue* pq)//取队头的数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//判断数据不能为空，为空则没有数据就会报错
	return pq->head->data;
}


QDataType QueueBack(Queue* pq)//取队尾的数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//判断数据不能为空，为空则没有数据就会报错
	return pq->tail->data;
}
```

## 统计队列元素个数
```C
int QueueSize(Queue* pq)//统计队列的数据个数
{
	assert(pq);
	int n = 0;
	QueueNode* cur = pq->head;
	while (cur != 0)
	{
		n++;
		cur = cur->next;
	}
	return 0;
```

## 队列元素的遍历，出数据
```C
void QueuePint(Queue* pq)
{
	assert(pq);
	printf("队列先进先出：");
	while (!QueueEmpty(pq))//队列不为空，证明有数据
	{
		QDataType value = QueueFront(pq);//取头数据
		printf("%d->", value);
		QueuePop(pq);
	}
	printf("\n");
}
```

# 队列总模块函数
## 声明函数(Queue.h)
```c
#pragma once
#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>
#include <assert.h>
#include <stdbool.h>

typedef int QDataType;

typedef struct QueueNode
{
	struct QueueNode* next;//递归的思路(单链表思路相同)
	QDataType data;
} QueueNode;

typedef struct Queue
{
	QueueNode* head;//指向头结构体的指针
	QueueNode* tail;//指向尾结构体的指针
}Queue;

void QueueInit(Queue* pq);//初始化
void QueueDestroy(Queue* pq);//销毁

void QueuePush(Queue* pq, QDataType x);//插入数据
void QueuePop(Queue* pq);//删除数据

QDataType QueueFront(Queue* pq);//取队头的数据
QDataType QueueBack(Queue* pq);//取队尾的数据
int QueueSize(Queue* pq);//统计队列的数据个数

bool QueueEmpty(Queue* pq);//判断队列是否为空

void QueuePint(Queue* pq);//队列数据的打印
```

## 住函数(Text.c)
```c
#define _CRT_SECURE_NO_WARNINGS 1
#include "Queue.h"


int main()
{
	Queue q;
	QueueInit(&q);
	QueuePush(&q, 1);//插入
	QueuePush(&q, 2);//插入
	QueuePush(&q, 3);//插入
	QueuePush(&q, 4);//插入
	QueuePush(&q, 5);//插入

	QueuePop(&q);
	printf("队头数据：%d \n", QueueBack(&q));//为数据
	printf("队尾数据：%d \n", QueueFront(&q));//头数据

	QueuePint(&q);
	return 0;
}
```

## 函数定义(Queue.c)
```C
#define _CRT_SECURE_NO_WARNINGS 1
#include "Queue.h"

void QueueInit(Queue* pq)//初始化
{
	assert(pq);
	pq->head = NULL;
	pq->tail = NULL;
}


void QueueDestroy(Queue* pq)//销毁
{
	assert(pq);
	QueueNode* cur = pq->head;
	while (cur != NULL)
	{
		QueueNode* next = cur->next;
		free(cur);
		cur = next;
	}
	pq->tail = pq->head = NULL;
}

void QueuePush(Queue* pq, QDataType x)//插入数据
{
	assert(pq);
	QueueNode* newnode = (QueueNode*)malloc(sizeof(QueueNode));
	newnode->data = x;
	newnode->next = NULL;
	if (pq->head == NULL)
	{
		pq->head = pq->tail=newnode;
	}
	else
	{
		pq->tail->next = newnode;
		pq->tail = newnode;
	}
}

void QueuePop(Queue* pq)//删除数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//bool和assert的联合运用，当boo为真是因为有!则assert为假
	QueueNode* next = pq->head->next;
	free(pq->head);
	pq->head = next;
	if (QueueEmpty(pq))
	{
		pq->tail = NULL;
	}
}

bool QueueEmpty(Queue* pq)//判断队列是否为空
{
	assert(pq);
	return pq->head == NULL;//如果结果成立，则传的是bool类型的真(Ture),为假的化就是传flase
}


QDataType QueueFront(Queue* pq)//取队头的数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//判断数据不能为空，为空则没有数据就会报错

}


QDataType QueueBack(Queue* pq)//取队尾的数据
{

	assert(!QueueEmpty(pq));//判断数据不能为空，为空则没有数据就会报错

}

	assert(pq);
	int n = 0;
	QueueNode* cur = pq->head;
	while (cur != 0)
	{
		n++;
		cur = cur->next;
	}
	return n;
}

{

		QDataType value = QueueFront(pq);//取头数据
		printf("%d->", value);
		QueuePop(pq);
	}
```
