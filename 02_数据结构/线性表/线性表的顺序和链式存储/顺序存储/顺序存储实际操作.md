```c
//线性表的初始化
void  SeqListInit(SL* ps)
{
	ps->a = NULL;
	ps->size = ps->capacity = 0;
}

//打印
void SeqListPrint(SL* ps)
{
	for (int i = 0; i < ps->size; i++)
	{
		printf("%d ", ps->a[i]);
	}
	printf("\n");
}

//检查容量
void SeqListCheckCapacity(SL* ps)
{
	//如果没有空间或者空间不足，那么我们就扩容
	if (ps->size == ps->capacity)
	{
		//下面使用的是三目运算符
		int newcapacity = ps->capacity == 0 ? 4 : ps->capacity * 2;//*2比较合理
		//根据realloc的特点（当原来的空间是空指针是，realloc的行为就相当于malloc）
		SLDataType* tmp = (SLDataType*)realloc(ps->a, newcapacity*sizeof(SLDataType));
		if (tmp == NULL)
		{
			printf("realloc fail\n");
			//异常exit终止程序
			exit(-1);//正常情况下是返回0，但是异常情况是-1
		}
		ps->a = tmp;
		ps->capacity = newcapacity;
	}
}
```

text.c
```c
 #define _CRT_SECURE_NO_WARNINGS 1
#include "Seqlist.h"

void TestSeqList()
{
	SL s1;
	SeqListInit(&s1);//线性表的初始化
	SeqListPushBack(&s1,1);//尾插
	SeqListPushBack(&s1, 2);
	SeqListPushBack(&s1, 3);
	SeqListPushBack(&s1, 4);
	SeqListPushBack(&s1, 5);
	printf("从尾部插入数据：");
	SeqListPrint(&s1);//打印

	SeqListPopFront(&s1);//尾删除
	printf("从尾部删除数据：");
	SeqListPrint(&s1);//打印

	SeqListPushFront(&s1,10);//头插入
	printf("从头部插入10：");
	SeqListPrint(&s1);//打印

	SeqListPopBack(&s1);//头删除
	printf("从头部删除：");
	SeqListPrint(&s1);//打印

	SeqListInsert(&s1, 2, 88);//指定位置下标插入
	printf("插入第3个数据：");
	SeqListPrint(&s1);//打印
	

	SeqListErase(&s1,3);//指定位置删除
	printf("删除第4个数据：");
	SeqListPrint(&s1);//打印

	SeqListFind(&s1, 88);//查找元素88
	

	SeqListDestory(&s1);//数据销毁
}



int main()
{
	TestSeqList();
	return 0;
}
```
