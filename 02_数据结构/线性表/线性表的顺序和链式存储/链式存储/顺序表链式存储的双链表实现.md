首先要明白的是要定义一个什么样的双链表？本次定义的是一个带哨兵位的双向循环链表。

# 结构体的定义

```c
typedef int LTDateType;
typedef struct ListNode
{
	LTDateType date;
	struct ListNode* next;//结构体的尾
	struct ListNode* prev;//结构体的头
}LTNode;
```

# 函数的定义
## 打印
```C
void ListPrint(LTNode* phead)//打印
{
	assert(phead);
	LTNode* cur = phead->next;
	while (cur != phead)
	{
		printf("->%d", cur->date);
		cur = cur->next;
	}
}
```
这里会存在一个错误的编写：这种编写会使得cur（地址）等于phead，这样的编写会造成一个死循环，``cur->next=phead其实等同与cur->next=cur或者phead->next=phead这样一个错误的编写``
```C
void ListPrint(LTNode* phead)//打印
{
	assert(phead);
	LTNode* cur = phead;
	while (cur->next = phead)
	{
		printf("->%d", cur->date);
		cur = cur->next;
	}
}
```

## 创建一个新的结构体准备插入
```c
LTNode* BuyListNode(LTDateType x)//创建一个新的结构体
{
	LTNode* newnode = (LTNode*)malloc(sizeof(LTNode));
	assert(newnode);
	newnode->date = x;
	newnode->next = NULL;
	newnode->prev = NULL;
}
```

## 头插入
双向循环双链表的结构不需要判断是否为第一个(头)，因为头是带哨兵位的头的。
```C
void ListPushFront(LTNode* phead, LTDateType x)//头插入
{
	assert(phead);
	LTNode* newnode = BuyListNode(x);
	LTNode* next = phead->next;
	phead->next = newnode;
	newnode->prev = phead;

	newnode->next = next;
	next->prev = newnode;
}
```

## 哨兵位头节点的创建
这里并未对``phead->date``进行初始化，应为在这个双向循环双链表中头节点的值可以是任何值。
```C
LTNode* ListInit()//哨兵位头节点的创建
{
	LTNode* phead = (LTNode*)malloc(sizeof(LTNode));
	assert(phead);
	phead->next = phead;
	phead->prev = phead;
	return phead;
}
```

## 头删除
```C
void ListPopFront(LTNode* phead)//头删除
{
	assert(phead&&phead->next!=phead);
	LTNode* next = phead->next;
	LTNode* nextNext = next->next;
	phead->next = nextNext;
	nextNext->prev = phead;
	free(next);
	next = NULL;
}
```

### 删除pos位置的元素
```C
void ListErase(LTNode* pos)//删除pos位置的元素
{
	assert(pos);
	LTNode* PosPrev = pos->prev;
	LTNode* PosNext = pos->next;
	PosPrev->next = PosNext;
	PosNext->prev = PosPrev;
}
```

## 销毁数据
这里这个哨兵为的头节点是没有销毁的，但是在主函数(main)里面会进行处理
```C
void ListDestory(LTNode* phead)//销毁数据
{
	assert(phead);
	LTNode* cur = phead->next;
	while (cur != phead)
	{
		LTNode* next = cur->next;
		free(cur);
		cur = next;
	}
}
```

```C
int main()

	ListPushFront(plist, 2);//头插入

	ListPushFront(plist, 4);//头插入
	ListPushFront(plist, 5);//头插入
	ListPrint(plist);

	ListPopFront(plist);//头删除
	ListPrint(plist);

	ListPushBack(plist,200);//尾插入
	ListPrint(plist);

	ListPopBack(plist);//尾删除
	ListPrint(plist);

	ListBack(pos,222);//在pos前面位置插入
	ListPrint(plist);

	ListErase(pos);//删除pos位置的元素
	ListPrint(plist);
	return 0;
}
```
