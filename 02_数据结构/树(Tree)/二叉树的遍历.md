> [!tip] 前序/中序/后序的递归结构遍历：
> 是根据访问结点操作发生位置命名 
> 1. NLR：前序遍历(Preorder Traversal 亦称先序遍历)——访问根结点的操作发生在遍历其左右 子树之前。 **(根  左子树   右子树)**
> 2. LNR：中序遍历(Inorder Traversal)——访问根结点的操作发生在遍历其左右子树之中 （间）。 **( 左子树   根  右子树)**
> 3. LRN：后序遍历(Postorder Traversal)——访问根结点的操作发生在遍历其左右子树之后。  **(左子树    右子树    根)**
>> 注意 
前序/中序/后序的遍历也叫做**深度优先遍历**

> 1. 上一层出的时候带下一层节点进
> 2. 同时在两个不同的结构里面存储二叉树的数据需要对数据进行**正确定义**
>>[!tip] 注：
>>用到的队列结构的使用笔记中用到的一样，只是在数据的定义上面有不同

### 队列的结构体和递归的二叉树如何一直存储同一颗二叉树
```C
//二叉树的递归(深度优先)结构体
typedef char BTDataType;
typedef struct BinaryTreeNode 
{
	struct BinaryTreeNode* left;
	struct BinaryTreeNode* right;
	BTDataType data;
}BTNode;

//typedef char QDataType;
typedef BTNode* QDataType;
typedef struct QueueNode
{
	struct QueueNode* next;//递归的思路(单链表思路相同)
	QDataType data;
} QNode;

typedef struct Queue
{
	QNode* head;//指向头结构体的指针
	QNode* tail;//指向尾结构体的指针
}Queue;

```

队列里面不能存具体的数值(A)，因为我们还要沿着向二叉树下面的节点进行遍历，存储了A以后把A拿出来后就不能找后面的数据了，所以队列这里要存储**节点的指针**``typedef BTNode* QDataType;``QdataType存储了BTNode这个节点的指针。

### 层次遍历的代码
```C
void LevelOrder(BTNode* root)//层次遍历
{
	//核心思路：上一层出的时候带下一层节点进
	Queue q;
	QueueInit(&q);
	if (root)
	{
		QueuePush(&q, root);
	}
	printf("\n层次遍历-广度优先:");
	while (!QueueEmpty(&q))
	{
		BTNode* front = QueueFront(&q);
		QueuePop(&q);
		printf("%c ", front->data);
		if (front->left != NULL)
		{
			QueuePush(&q, front->left);
		}
		if (front->right != NULL)
		{
			QueuePush(&q, root->right);
		}
	}
	printf("\n");
	QueueDestroy(&q);
}
```

## 函数的定义-Tree.c
```C
#define _CRT_SECURE_NO_WARNINGS 1
#include "Tree.h"

void PrevOrder(BTNode* root)//前序
{
	if (root == NULL)
	{
		printf("NULL ");
		return;
	}

	printf("%c ", root->data);
	PrevOrder(root->left);
	PrevOrder(root->right);
}

void InOrder(BTNode* root)//中序
{
	if (root == NULL)
	{
		printf("NULL ");
		return;
	}

	PrevOrder(root->left);
	printf("%c ", root->data);
	PrevOrder(root->right);
}

void PostOrder(BTNode* root)//后续
{

	if (root == NULL)
	{
		printf("NULL ");
		return;
	}

	PrevOrder(root->left);
	PrevOrder(root->right);
	printf("%c ", root->data);
}

int TreeSize(BTNode* root)//二叉树节点的个数
{
	return root == NULL ? 0 : TreeSize(root->left) + TreeSize(root->right) + 1;
}

int TreeLeafSize(BTNode* root)//叶子节点个数
{
	if (root == NULL) return 0;
	if (root->left == NULL && root->right == NULL) return 1;
	return TreeLeafSize(root->left) + TreeLeafSize(root->right);
}


void LevelOrder(BTNode* root)//层次遍历
{
	//核心思路：上一层出的时候带下一层节点进
	Queue q;
	QueueInit(&q);
	if (root)
	{
		QueuePush(&q, root);
	}
	printf("\n层次遍历-广度优先:");
	while (!QueueEmpty(&q))
	{
		BTNode* front = QueueFront(&q);
		QueuePop(&q);
		printf("%c ", front->data);
		if (front->left != NULL)
		{
			QueuePush(&q, front->left);
		}
		if (front->right != NULL)
		{
			QueuePush(&q, root->right);
		}
	}
	printf("\n");
	QueueDestroy(&q);
}

void QueueInit(Queue* pq)//初始化
{
	assert(pq);
	pq->head = NULL;
	pq->tail = NULL;
}

void QueuePush(Queue* pq, QDataType x)//插入数据
{
	assert(pq);
	QNode* newnode = (QNode*)malloc(sizeof(QNode));
	newnode->data = x;
	newnode->next = NULL;
	if (QueueEmpty(pq))//判断结构体QueueNode数据是否为空
	{
		pq->head = pq->tail = newnode;
	}
	else
	{
		pq->tail->next = newnode;
		pq->tail = newnode;
	}
}

bool QueueEmpty(Queue* pq)//判断队列是否为空
{
	assert(pq);
	return pq->head == NULL;//如果结果成立，则传的是bool类型的真(Ture),为假的化就是传flase
}


QDataType QueueFront(Queue* pq)//取队头的数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//判断数据不能为空，为空则没有数据就会报错
	return pq->head->data;
}

void QueuePop(Queue* pq)//删除数据
{
	assert(pq);
	assert(!QueueEmpty(pq));//bool和assert的联合运用，当boo为真是因为有!则assert为假
	QNode* next = pq->head->next;
	free(pq->head);
	pq->head = next;
	if (QueueEmpty(pq))
	{
		pq->tail = NULL;
	}
}

void QueueDestroy(Queue* pq)//销毁
{
	assert(pq);
	QNode* cur = pq->head;
	while (cur != NULL)
	{
		QNode* next = cur->next;
		free(cur);
		cur = next;
	}
	pq->tail = pq->head = NULL;
}
```
