# 数组指针
**主语是指针，本质是指针。**
```c
int mian()
{
	int* p = NULL;
	//p 是整型指针-指向整型的指针-可以存放整型的地址
	
	char* pc = NULL;
	//pc 是字符指针-指向字符的指针-可以存放字符的地址
	
	//数组指针-指向数组的指针-存放数组的指针
	int arr[10] = {0};
	//arr- 首元素地址
	//&arr[0]- 首元素地址
	//&arr - 数组的地址

	int arr[10] = {1,2,3,4,5};
	int(*p)[10] = &arr;//数组的地址要存起来
	//上面的p就是数组指针
	
	return 0;
}
```

那么判断一下下面那个是数组指针？
```c
int *p1[10];//指针数组
int (*p2)[10];//数组指针
```

如何正确的编写数组指针：
```c
int main()
{
	char* arr[5];
	char* (*p)[5] = &arr；

	int brr[5] = {0};
	int (*p)[5] = &brr;
	return 0;
}
```

数组指针的使用：**一般使用到二维数组以上。**

*首先要明白二维数组在我们取地址时和一维数组的不同之处*
![[二位数组在内存中的存储#^f5eed4]]

```c
#include <stdio.h>
void plent(int(*p)[3], int x, int y)
{
	int i = 0;
	for (i = 0; i < x; i++)
	{
		int j = 0;
		for (j = 0; j < y; j++)
		{
			printf("%d ", *(*(p + i) + j));
		}
		printf("\n");
	}
}

int main()
{
	int arr[2][3] = { {1,2,3},{2,3,4} };
	plent(arr, 2, 3);
	return 0;
}
```

```c
printf("%d ",*(*(p+i)+j));
这里的打印编写也可以有不同的编写方法：
(*(p+i))[j]
*(p[i]+j)
p[i][j]

可以通过下下面的代码，对上面的转换有了解：
int main()
{
	int arr[3] = {1 , 2 , 3 };
	int i = 0;
	int* p = &arr;
	for( i = 0 ; i < 3 ; i ++)
	{
		printf("%d ",*(p+i));
		//是不是也可以编写成
		printf("%d ",p[i]);
		printf("%d ",arr[i]);
		printf("%d ",*(arr+i));
		//arr[i] == *(arr+i) == *(p+i) == p[i]
	}
	return 0;
}

*(p+i) == p[i]
```

# 指针数组
**主语是数组，其实是一个存放指针的数组。**
```c
int arr[10];//整型数组
char ch[5] = {0};//字符数组

int* arr1[10];//整型指针数组-存放整型指针的数组
char* arr2[4];//一级指针的数组
char** arr3[5];//二级指针的数组
```


使用的方法：
```c
#include <stdio.h>
int main()
{
	int arr1[] = { 1,2,3,4,5 };
	int arr2[] = { 2,3,4,5,6 };
	int arr3[] = { 3,4,5,6,7 };
	int i = 0;
	int* p[3] = {arr1,arr2,arr3};
	//指针 p [0] 存放的是 arr1 首元素的地址
	//指针 p [1] 存放的是 arr2 首元素的地址
	//指针 p [2] 存放的是 arr3 首元素的地址
	
	for (i = 0; i < 3; i++)
	{
		int j = 0;
		for (j = 0; j < 5; j++)
		{
			printf("%d ", *(p[i] + j));//p[i]+j 是对首地址后面地址的访问。
		}
		printf("\n");
	}
	return 0;
}
```
