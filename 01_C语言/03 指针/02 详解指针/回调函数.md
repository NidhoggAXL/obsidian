> [!tip] 定义
> 回调函数就是一个通过函数指针调用的函数。
> 如果你把函数的指针（地址)作为参数传递给另一个函数，当这个指针被用来调用其所指向的函数时，我们就说这是回调函数。回调函数不是由该函数的实现方直接调用，而是在特定的事件或条件发生时由另外的一方调用的，用于对该事件或条件进行响应。

使用例子：
```C
#define _CRT_SECURE_NO_WARNINGS 1
#include <stdio.h>

void menu()
{
	printf("*******************\n");
	printf("***1.Add  2.Sub****\n");
	printf("******0.exit*******\n");
	printf("*******************\n");
}

int Add(int x, int y)
{
	return x + y;
}

int Sub(int x, int y)
{
	return x - y;
}

void Ccol(int (*p)(int, int))
{
	int x, y;
	printf("请输入操作数->");
	scanf("%d%d", &x, &y);
	printf("%d\n",p(x, y));
}

int main()
{
	int input, x, y;
	do
	{
		menu();
		printf("请选择操作->");
		scanf("%d", &input);
		switch (input)
		{
		case 1:
			Ccol(Add);
			/*printf("请输入操作数->");
			scanf("%d%d", &x, &y);
			printf("%d\n",Add(x, y));*/
			break;
		case 2:
			Ccol(Sub);
			/*printf("请输入操作数->");
			scanf("%d%d", &x, &y);
			printf("%d\n",Sub(x, y));*/
			break;
		case 0:
			printf("exit\n");
			break;
		default:
			printf("输入错误\n");
		}
	} while (input);
	return 0;
}
```
