> [!tip] 定义
> ```
> int ( *p [ ] ) ( );
> p先和 [ ] 结合,[ ] 的优先级高于 * ，说明 p 是数组，
> 数据的内容是 int ( * ) ( )
> ```

```c
int Add(int a, int b)
{
	return a + b;
}
int Sub(int a, int b)
{
	return a - b;
}
int Mul(int a, int b)
{
	return a * b;
}
int Div(int a, int b)
{
	return a / b;
}

int main()
{
	//int* arr[5];指针数组
	int (*p[4])(int, int) = {Add,Sub,Mul,Div};//函数指针数组
	int i;
	for (i = 0; i < 4; i++)
	{
		printf("%d\n", p[i](2, 3));
	}
	return 0;
}
```

函数指针数组的用途：转移表
计算器
```C
void menu()
{
	printf("*******************\n");
	printf("***1.Add  2.Sub****\n");
	printf("******0.exit*******\n");
	printf("*******************\n");
}

int Add(int x, int y)
{
	return x + y;
}

int Sub(int x, int y)
{
	return x - y;
}
int main()
{
	int input, x, y;
	do
	{
		menu();
		printf("请选择操作->");
		scanf("%d", &input);
		//aprr 是一个函数指针-转移表
		int (*aprr[])(int, int) = {0,Add , Sub};
		if (input > 0 && input < 3)
		{
			printf("请输入操作数->");
			scanf("%d%d", &x, &y);
			printf("%d\n", aprr[input](x, y));
		}
		else if (input == 0)
		{
			printf("退出系统\n");
		}
		else
		{
			printf("输入错误\n");
		}
	} while (input);
	return 0;
}
```
