**注意一定要进行初始化**

创建了俩个数组，如果依次都进行初始化就会麻烦，不妨设置一个数组board来接收这两个数组，同时因为要打印出来0和``*``号也可以设置一个元素char set来接收这两个符号。
```c
void InisBoard(char board[ROWS][COLS], int rows, int cols, char set)
{
	int i = 0;
	int j = 0;
	for (i = 0; i < rows; i++)
	{
		for (j = 0; j < cols; j++)
		{
			board[i][j] = set;
		}
	}
```

打印棋盘，为了方便玩家的游戏，在开头要设立一个行数和列数。**特别注意：这里打印出来的行数和列数是不在我们布置的扫雷表格里面，这里传进行的ROWS和COLS是``11*11``的其中的边框是没有打印出来的。**
```c
void DisplyBoard(char board[ROWS][COLS], int row, int col)
{
	int i = 0;
	int j = 0;
	//打印行数
	for (i = 0;i<=col;i++)
	{
		printf("%d ", i);
	}
	printf("\n");
	for (i = 1;i<=row;i++)
	{
		printf("%d ", i);//打印列数
		for (j = 1;j<=col;j++)
		{
			printf("%c ", board[i][j]);
		}
		printf("\n");
	}
}
```

在mine中对雷进行布置，因为要随机进行雷的布置，要使用到随机函数[[rand和srand]]和[[time]]来进行随机雷的布置。雷用**字符1**来表示，不是数字1.
```c
雷的个数
#define THUNDET 10
```

```c
void SetMinr(char board[ROWS][COLS], int row, int col)
{
	int lei = THUNDET;
	while (lei)
	{
		int x = rand() % row + 1;
		int y = rand() % col + 1;
		if (board[x][y] == '0')
		{
			board[x][y] = '1';
			lei--;
		}
	}
}
```

玩家扫雷后要对周围的雷进行一个扫描。

当周围进行雷的扫描的时候，要计算雷的个数，并在排查出雷的信息中show表示出来。这里就要计算雷的个数，我们要知道这里的雷是*字符1*来表示的，但是我们这么才嫩计算出雷的个数呢？

``我们知道在刚开始还没有进行布置雷的时候，我们mine是存储'0'的，后面进行了雷的布置后就存在了雷('1')在里面。这里'1'-'0'=0 '2'-'1'=2这里就运用到了``[[ASCII表值]]中字符所对应的大小来表示了。
```c
//扫雷
void FindMine(char mine[ROWS][COLS],char show[ROWS][COLS], int row, int col)
{
	int x = 0;
	int y = 0;
	int win = 0;
	//9*9-10=71
	while (win<row*col- THUNDET)
	{
		printf("请输入扫雷的坐标:");
		scanf("%d%d", &x, &y);
		if (x >= 1 && x <= row && y >= 1 && y <= col)//坐标正确
		{
			//1.踩雷
			if (mine[x][y] == '1')
			{
				printf("很遗憾，你被炸死了\n");
				break;
			}
			//2.没有踩雷
			else
			{
				//计算x y周围有几个雷
				int count = get_mine_count(mine, x, y);
				show[x][y] = count + '0';//把count强制转话我char类型
				DisplyBoard(show, ROW, COL);//打印排查雷的信息
				win++;
			}
		}
		else//坐标错误
		{
			printf("输入坐标错误,");
		}
	}
	if (win == row*col- THUNDET)
	{
		printf("\n");
		printf("恭喜你，排雷成功\n");
		DisplyBoard(mine, ROW, COL);
	}
}

//对排查位置周围的雷进行统计
int get_mine_count(char mine[ROWS][COLS], int x, int y)
{
	return
		mine[x - 1][y] +
		mine[x - 1][y - 1] +
		mine[x][y - 1] +
		mine[x + 1][y - 1] +
		mine[x + 1][y] +
		mine[x + 1][y + 1] +
		mine[x][y + 1] +
		mine[x - 1][y + 1] - 8 * '0';
}
```

```c
#define _CRT_SECURE_NO_WARNINGS 1
#include <stdio.h>
#include "game.h"

void game()
{
	//雷的信息存储
	//1.布置好雷的信息
	char mine[ROWS][COLS] = { 0 };//11*11
	//2.排查出雷的信息
	char show[ROWS][COLS] = { 0 };
	//初始化
	InisBoard(mine, ROWS, COLS, '0');
	InisBoard(show, ROWS, COLS, '*');
	//打印
	//DisplyBoard(mine, ROW, COL);
	printf("\n");
	DisplyBoard(show, ROW, COL);
	printf("\n");
	//布置雷
	SetMinr(mine, ROW, COL);
	DisplyBoard(mine, ROW, COL);
	//扫雷
	FindMine(mine, show, ROW, COL);
}

void menu()
{
	printf("**********************\n");
	printf("*******1. play********\n");
	printf("*******0. exit********\n");
	printf("**********************\n");
}

void text()
{
	int input = 0;
	srand((unsigned int)time(NULL));
	do
	{
		menu();
		printf("请确定游戏的开始或者结束");
		scanf("%d", &input);
		switch (input)
		{
		case 1:
			printf("游戏开始\n");
			game();
			break;
		case 0:
			printf("游戏结束\n");
			break;
		default:
			printf("输入错误，请重新输入\n");
			break;
		}
	} while (input);
}

int main()
{
	text();
	return 0;
}
```
