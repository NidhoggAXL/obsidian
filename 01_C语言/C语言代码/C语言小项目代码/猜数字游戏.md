1. 原理：
	* 电脑会生成一个随机数
		* 那[[时间戳]]来设置随机数的生成起始点。
		* 生成随机数运用到的函数：[[rand和srand]]和[[time]]。
	* 猜数字
		* 利用if循环来减小范围来更加的抄到数字。
2. 代码：
```c
#include <stdio.h>
#include <time.h>
#include <stdlib.h>

void menu()
{
	printf("*******************\n");
	printf("***1.开始 2.结束***\n");
	printf("*******************\n"); 
}

void game()
{
	int ret = 0;
	int guess = 0; 
	ret = rand()%100+1;//生成1~100之间的随机数 
	while(1)
	{
		printf("请输入数字：");
		scanf("%d",&guess);
		if(guess>ret)
		{
			printf("猜大了\n");
		}
		else if(guess<ret)
		{
			printf("猜小了\n"); 
		}
		else if(guess==ret)
		{
			printf("恭喜你,请对了\n");
			break ;
		} 
	}	
 } 
 
 int main()
 {
 	
 	int input = 0;
 	srand((unsigned int)time(NULL)); 
 	do
 	{
 		menu ();
 		printf("请选择>:");
 		scanf("%d",&input);
		switch(input)
 		{
 			case 1:
 				game();//猜数字有戏开始
 				break;
			case 0:
				printf("退出游戏\n");
				break;
			default:
				printf("选择错误\n");
				break ; 
		 }	
	 }while(input);
 	return 0;
 }
 
```
