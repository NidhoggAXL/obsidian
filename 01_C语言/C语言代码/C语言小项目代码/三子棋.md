前言：和电脑进行三子棋游戏，这里要使用代码的模块化来实现。需要创建两个源文件(game.c和text.c),一个头文件(game.h)

三子棋是一个``3*3``的表格，这里可以创建一个二维数组来存储数据。方便以后对数组的改变，利用动态变量来进行初始化(ROW.COL)。
```c
board[ROW][COL]={0};
```
*要注意在传输过程中的大小写。define定义使必须是大写*。

棋盘的初始化：**数组一定要进行初始化**但是为了方便观察，使三子棋盘里面看起来没有什么，数组里面就进行初始化为里面全部是空格。
```c
void Intboard(char board[ROW][COL], int row, int  col)
{
	int i = 0;
	int j = 0;
	for (i = 0; i < row; i++)
	{
		for (j = 0; j < col; j++)
		{
			 board [i][j] = ' ';
		}
	}
}
```

电脑下棋：
电脑进行下棋时，如果进行随机下棋，那么就需要一个随机数字来进行下棋[[rand和srand]]和[[time]]来进行随机数的开始。我们这里进行的是一个3乘3的一个棋盘，因为是那数组来进行数据的存储，所以下标从0开始。对随机数取余数就可以得到（0、1、2）来进行下棋。
```c
void ComputerMove(char board[ROW][COL], int row, int col)//电脑下棋
{
	int x = 0;
	int y = 0;
	printf("电脑下棋：>\n");
	while (1)
	{
		x = rand() % row;//横坐标随机数取余数为0 1 2
		y = rand() % col;//列坐标随机数取余数为0 1 2
		if (board[x][y] == ' ')
		{
			board[x][y] = '#';
			break;
		}
	}
}
```

对输赢结果的表示：

* '*'表示玩家下的棋子，同时返回值也表示玩家赢。

* '#'表示电脑下载的棋子，同时返回值也表示电脑赢。

* '-'表示棋盘没有下完，还要继续下棋。

* '='表示棋盘下完，平局。

```c
void game()//游戏的整个实现
{
	char ret = 0;
	char board[ROW][COL] = { 0 };//数组的创建
	Intboard(board, ROW, COL);//棋盘的初始化为空格
	Displayboard(board, ROW, COL);//棋盘的打印
	while (1)
	{
		PlayerMove(board, ROW, COL);//玩家下棋
		Displayboard(board, ROW, COL);//棋盘的打印
		printf("\n");
		ret = IsWin(board,ROW,COL);
		if (ret != '-')
		{
			break;
		}
		ComputerMove(board, ROW, COL);//电脑下棋
		Displayboard(board, ROW, COL);//棋盘的打印
		ret = IsWin(board,ROW,COL);
		if (ret != '-')
		{
			break;
		}
	}
	if (ret == '*')printf("玩家赢\n");
	if (ret == '#')printf("电脑赢\n");
	if (ret == '=')printf("平局\n");
}
```

判断是否连接3个棋子（判断输赢）：
* 先从三横和三列来判断
* 在从两个对角线来判断
* 最后还要判断是否下满棋盘
```c
//判断是否下满棋盘
int  IsPlay(char board[ROW][COL], int row, int col)
{
	int i = 0;
	int j = 0;
	for (i = 0; i < row; i++)
	{
		for (j = 0; j < col; j++)
		{
			if (board[i][j] == ' ')
			{
				return 0;
			}
		}
	}
	return 1;
}

//判断输赢
char IsWin(char board[ROW][COL], int row, int col)
{
	int i = 0;
	for (i = 0; i < row; i++)
	{
		//三横
		if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][0] != ' ')
		{
			return board[i][0];
		}
		//三列
		if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[0][i] != ' ')
		{
			return board[0][i];
		}
	}
	//两个对角线
	if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[1][1] != ' ')
	{
		return board[1][1];
	}
	if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[1][1] != ' ')
	{
		return board[1][1];
	}
	if (1 == IsPlay(board, ROW, COL))
	{
		return '=';
	}
	return '-';
}
```
