* 1-1/2+1/3-1/4+......+1/(n-1)+1/n 和的计算。
```c
#include <stdio.h>
int main()
{
	int i =0;
	double sum = 0.0;
	int j = 1;
	for(i = 1;i<=100;i++)
	{
		sum+=j*1.0/i;
		j = -j;	
	} 
	printf("%lf",sum);
	return 0;
}
```
