#include<stdio.h>
int main(int argc,char *argv[])
{
 char *s[]={
  "January","February","March","April","May",
  "June","July","August","September","October",
  "November","December"
 };
 printf("请输入月份数,输入-1结束查询:\n");
 int idx;
 
 while(idx != -1)
 {
  scanf("%d",&idx);
  if(idx==-1){
   printf("您结束了查询。");
  }else if(idx>12 || idx<=0)
  {
   printf("您输入的月份数有误，请重新输入。");
  }else{
    printf("%d月对应的英文月名为%s\n",idx,s[idx-1]);
  }
 

 }
}
