	* 三月：==31==

	* 五月：==31==
	* 六月：**30**
	* 七月：==31==

	* 十一月：**30**
	* 十二月：==31==


具体说，一年中有*7个月有31天，4个月有30天，2月份有28天，闰年的2月份有29天。*



* 月份统一后在分情况计算。
	* *要好好学习这里的思路，找到中间值来计算。设立断点找到方法。*
```c
#include <stdio.h>
int main()
{
	int when(int year, int month, int day);
	int year,month,day;
	printf("请依次输入年月日：");
	scanf("%d%d%d", &year, &month, &day);
	printf("该日是%d年的第%d天！", year, when(year, month, day));
	//when(year,month,day)这里就是when_day的返回值。
	return 0;
}

int when(int year, int month, int day){
	int when_day;
	when_day = (month-1)*30 + day - 2;
	 // 这是非闰年,且每个月只计算了30天; 
	if((year%4==0 && year%100!=0) || year%400==0){
	// 判断这一年是不是闰年
	printf("%d年是闰年，所以二月有29天！\n",year); 
		if(month>2){
		when_day += 1;
		}
	}
 	if(month<=7){
 	//利用7月为一个断开点，七月上下个有3个月的31天。
  	when_day += month/2;
 	}
	else{
  	when_day += month/2+1;
 	}
 	return when_day; 
}
```

```c
#include <stdio.h>
#include <stdlib.h>

int main()

}

int count(int y ,int m ,int d)
{
	int n = 0;
	int j = 0;
	int i = 0;
	int arr[]={0,31,28,31,30,31,30,31,31,30,31,30};
	//每个月份有多少天 ， 
	//一月的话就是0天，后面+几日。12月的话就只计算12月以前的天数  
	for(n = 0; j < m ;j++)
	{
		n = n + arr[j];
	}
	i = n + d;
	if(y%4==0&&y%100!=0||y%400==0)//判读闰年

		if(m>2)
		//如果时2月的话，就只过了1月份，所以这里不加 =

		}

}
```
