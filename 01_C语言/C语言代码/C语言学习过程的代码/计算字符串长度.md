 #指针
 
```c
#include <stdio.h>
#include <string.h>

int my_stelen(cha* str)//利用指针接收地址
{
	int count = 0;//临时变量
	while (*str != '\0')
	{
		count++;
		str++;//指针自加到a后面的字符b(进行循环)
	}
}
int main()
{
	cha arr[] = "abc";
	int len = my_strlen(arr);
	//arr是一个数组，数组传参，传过去的不是整个数组而是而是第一个元素的地址
	printf("%d\n", len);
}
```


#递归 不创建临时变量，求字符串的长度
```c
#include <stdio.h>
#include <string.h>

//递归方法，把大事化小
//my_strlen("bit");
//1 + my_strlen("it");
//1 + 1 + my_strlen("t");
//1 + 1 + 1 + my_stelen("");
//1 + 1 + 1 = 0
//3

int my_strlen(char* str)
{
	if (*str != '\0')//*srt为b
	{
		return 1 + my_strlen(str+1);//指针自加到b后面的字符i
	}
	else
		return 0;
}
int main()
{
	char arr[] = "abc";
	int len = my_strlen(arr);
	//arr是一个数组，数组传参，传过去的不是整个数组而是而是第一个元素的地址
	printf("%d\n", len);
}

```
