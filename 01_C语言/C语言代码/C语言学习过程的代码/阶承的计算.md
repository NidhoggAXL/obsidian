# 计算n的阶乘
利用循环
```c
#include <stdio.h>
int main()
{	
	int i = 0;
	int n = 0;
	int ret = 1;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		ret = ret * i;
	}
	printf("%d\n",ret);
	return 0;
 } 
 这里需要注意这里ret为整型，计算的值不能超过int整型值。
```

#递归 的利用
```c
#include <stdio.h>
int face(int n)
{
	if (n == 1)
	{
		return 1;
	}
	else
		return n * face(n - 1);
}
int main()
{
	int n = 0;
	scanf("%d", &n);
	int ret = face(n);
	printf("%d\n", ret);
	return 0;
}
```

# 计算n的阶乘和
```c
#include <stdio.h>
int main()
{	
	int i = 0 , j = 0;
	int n = 0;
	int ret = 1;
	int sum = 0;
	scanf("%d",&n);
	for(i =1 ;i<=n;i++)
	{
		ret = 1;
		for(j=1;j<=i;j++)
		{
		ret = ret * j;
		}	//这里计算第i的阶乘 
		sum = sum + ret ;
	 } //这里对i的n阶乘进行累加 
	printf("%d\n",sum);
	return 0;
 } 
```
