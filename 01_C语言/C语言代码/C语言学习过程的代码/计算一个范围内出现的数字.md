*  原理：含有某个数字的这能是在个位、十位、百位...中含有某个字。
# 查找1~100中含有9的数字，并且计算有有多少个九
```c
#include <stdio.h>
int main()
{
	int i = 0;
	int n = 0;
	printf("查找1~100中含有9的数，并且计算9的个数\n"); 
	printf("含有9的数字：");
	for (i = 1;i<=100;i++)
	{	
	
		if(i%10==9||i/10==9){
			printf("%d ",i);
			n ++;
		}
		//9 19 29 39 49 59 69 79 89(i%10==9)
		// 90 91 92 93 94 95 96 97 98 99(i/10==0)
		if(i%10==9&&i/10==9)
			n++;
	}//对 99 这个数字应为有俩个 9 9 但是循环的时候
	//就只对99进行了一次的 n++，所以后边在编写一个if对 n++.
	printf("\n");
	printf("含有9的个数：%d",n);
	return 0;
}
```
