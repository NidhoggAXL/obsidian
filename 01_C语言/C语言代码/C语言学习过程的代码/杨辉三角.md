* 杨辉三角：
![gh](https://raw.githubusercontent.com/AXLflechazoPN/Obsidian/main/2024/17052150710005r9hfu.png)



``` c
#include<stdio.h>
 //从第三行开始，除了对角线和第一列的值外 
// 所有的值=头顶上两个数值之和 a[i][j]=a[i-1][j-1]+a[i-1][j]
int main(){
	int a[10][10];
	int i , j ;
	for(i=0;i<10;i++){ 
	//这里赋值数列的开头与结尾为 1 
		a[i][i]=1;
		a[i][0]=1;
	}
	for(i=2;i<10;i++){ 
	 //第一行只有一个数 1 ，第二行的数为 1 1 
	 //所以从第3行开始。 
		for(j=1;j<i;j++){ 
		//这里根据杨辉三角的规律，j<i是对第二列
		//开始计算的这样就可以找高j<i这个规律 
			a[i][j]=a[i-1][j-1]+a[i-1][j];
		}
	}
	for(i=0;i<10;i++){ //要对第1~10行赋值
		for(j=0;j<=i;j++){ //对第1~10列赋值
			printf("%d ",a[i][j]);
		}
		printf("\n"); 
		 //每一行结束进行换行，这样才可以输出杨辉三角的样式。
	}
	return 0;
}
```
