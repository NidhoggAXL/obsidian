#递归
```c
#include <stdio.h>

void print(int n)
{
	if (n > 9)
	//递归的两个必要条件
//* 存在限制条件，当满足这个限制条件的时候，递归便不再继续。
	{
		print(n/10);
		//* 每次递归调用之后越来越接近这个限制条件。
	}
	printf("%d ", n % 10);
}
int main() {
	unsigned int num = 0;
	scanf("%d", &num);
	print(num);
	return 0;
}
```
