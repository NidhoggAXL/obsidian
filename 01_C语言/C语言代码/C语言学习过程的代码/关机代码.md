* 数字字符串停止关机[[shutdown]]
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h> 

int main()
{
	char input[10]={0};
	system("shutdown -s -t 60");
	//-s(关机)  -t(时间单位毫秒)
gain:
	printf("电脑将在60秒后关机，如果输入：我是猪，关机取消");
	printf("请输入："); 
	scanf("%s",&input);
	if(strcmp(input,"我是猪")==0)
	{
		system("shutdown -a");
		//-a(停止关机)
		printf("关机取消成功"); 
	}else
	{
		goto gain;
	}
	return 0;
}
```
