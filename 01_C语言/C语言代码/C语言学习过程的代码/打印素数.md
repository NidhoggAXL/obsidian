#自定义函数打印素数
```c
#include <stdio.h>
int is_prime (int n)
{
	int j = 0;
	for(j = 2;j<=n/2;j++)
	{
		if(n%j==0)
		{
			return 0;
		}
	 } 
	 return 1;
}

int main()
{
	int i = 0;
	printf("1~100的素数有：");
	for(i = 100;i<=200;i++)
	{
		if(is_prime(i) == 1)
			printf("%d ",i);
	}
	return 0;
}
```
