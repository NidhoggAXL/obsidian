```c
#include<stdio.h>
int main()
{
	int att[] = { 1,2,3,4,5,6 };
	int se = sizeof(att) / sizeof(att[0]);
	int i = 0;
	for (i = 0; i < se; i++)
	{
		printf("&att[%d]= %d\n", i,&att[i]);
	}
	return 0;
}
```
