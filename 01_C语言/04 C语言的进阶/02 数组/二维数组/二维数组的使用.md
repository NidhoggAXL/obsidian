二维数组的使用依然是通过下标来使用的。
```c
int main()
{
	int arr[3][4] = { {1,2,3},{4,5} };
	int i = 0, j = 0;
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 4; j++)
		{
			printf("%d ", arr[i][j]);
		}
		printf("\n");
	}
	return 0;
}
```
