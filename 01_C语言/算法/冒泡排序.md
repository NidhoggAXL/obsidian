前言：数组中存储元素，后面在排序。

思路：随着要比较元素的减少，每一次要进行的冒泡排序的趟数也在减少。

低效代码：
```c
#include<stdio.h>
void maopao(int arr[], int len)//要注意这里的[]号，不可以省略。
{
	int i = 0;
	for (i = 0; i < len - 1; i++)//确定冒泡排列趟数
	{
		int j = 0;
		for (j = 0; j < len - 1 - i; j++)//确定要比较元素的个数
		{
			if (arr[j] > arr[j + 1])
			{
				int tmp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = tmp;
			}
		}
	}

}
int main()
{
	int arr[] = { 9,8,7,6,5,4,3,2,1,0 };
	int len = sizeof(arr)/sizeof(arr[0]);
	maopao(arr, len);
	int i = 0;
	for (i = 0; i < len; i++)
	{
		printf("%d ", arr[i]);
	}
	return 0;
}
```

高效代码：
```c
#include<stdio.h>
void maopao(int arr[], int len)//要注意这里的[]号，不可以省略。
{
	int i = 0;
	for (i = 0; i < len - 1; i++)//确定冒泡排列趟数
	{
		int j = 0;
		int flag = 1;//假设这一趟要排序的数据是有序的
		for (j = 0; j < len - 1 - i; j++)//确定要比较元素的个数
		{
			if (arr[j] > arr[j + 1])
			{
				int tmp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = tmp;
				flag = 0;//要进行交换，则本趟排序的数据不完全有序
			}
		}
		if (flag == 1)
		{
			break;
		}
	}

}
int main()
{
	int arr[] = { 9,8,7,6,5,4,3,2,1,0 };
	int len = sizeof(arr)/sizeof(arr[0]);
	maopao(arr, len);
	int i = 0;
	for (i = 0; i < len; i++)
	{
		printf("%d ", arr[i]);
	}
	return 0;
}
```


再次优化： #回调函数  #qsort  #特殊指针类型void 
```C
#define _CRT_SECURE_NO_WARNINGS 1
#include <stdio.h>

int cmp_int(const void* e1, const void* e2)
{
	//用void接收，所以不可以直接解引用
	//（int*)强制类型转换我int*型后就可以解引用了
	return *(int*)e1 - *(int*)e2;
}

void Swap(char* buf1, char* buf2, int width)
{
	int i = 0;
	//下面循环实际上就是从根本的字节数上面进行交换
	//不同类型有不同的字节数
	//i<width 就是从第一个字节交换到第一个个宽度-width（单个元素的大小）
	for (i = 0; i < width; i++)
	{
		char tmp = *buf1;
		*buf1 = *buf2;
		*buf2 = tmp;
		buf1++;
		buf2++;
	}
}

void bubble_sort(void* base, int se, int width, int (*cmp) (void* e1, void* e2))
{
	int i, j;
	for (i = 0; i < se - 1; i++)
	{
		for (j = 0; j < se - 1 - i; j++)
		{
			//两个元素比较
			//if里面运用回调函数，回调到cmp_int
			if (cmp((char*)base + j * width, (char*)base + (j + 1) * width) > 0)
			{
				//交换
				Swap((char*)base + j * width, (char*)base + (j + 1) * width, width);
			}
		}
	}
}
void text1()
{
	int arr[5] = { 5,4,3,2,1 };
	int se = sizeof(arr) / sizeof(arr[0]);
	bubble_sort(arr, se, sizeof(arr[0]), cmp_int);
}



int main()
{
	text1();
	return 0;
}
```
