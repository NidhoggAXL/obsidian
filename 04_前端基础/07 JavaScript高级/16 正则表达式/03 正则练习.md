从数组每一个元素里面得到时间，使用正则表达式进行匹配：

```js
<script>
  //[]在正则表达式有特殊含义，要使用转移字符
  const timeRe1 = /\[\]/i
  
  //munute(分)只会是两位数 \d{2}
  //second(秒)只会是两位数 \d{2}
  //millisecond(毫秒)会是两位数或者三位数 \d{2,3}
  //. 特殊字符，要使用转移字符 \.qw
  const timeRe2 = /\[\d{2}:\d{2}\.\d{2,3}\]/i
  
  //后续要对分秒，装换为毫秒
  //毫秒要进行两位数或者三位数进行判断操作
  //使用捕获组进行分组
  const timeRe3 = /\[(\d{2}):(\d{2})\.(\d{2,3})\]/i

  //最终得到最后的正则表达式
  const timeRe = /\[(\d{2}):(\d{2})\.(\d{2,3})\]/i
</script>
```

获取分、秒、毫秒的时间数组：

```js
<script>

    const result = lineString.match(timeRe)
```

# 二、日期格式化
时间格式化:从服务器拿到时间戳，转成想要的时间格式

```js
<script>
  //获取本地时间戳
  const newdata = new Date().getTime()
  console.log(newdata)
  //1733231045322
  
  //根据时间帧得到 RFC2822时间格式
  const data = new Date(1733231045322)
  console.log(data)
</script>
```

```js
<script>
  function formatDate(time, formatDate) {
    const date = new Date(time)
    //正则表达式和时间值对应
    //所以放入到对象里面
    const obj = {
      "y+": date.getFullYear(),
      "M+": date.getMonth() + 1,
      "d+": date.getDate(),
      "h+": date.getHours(),
      "m+": date.getMinutes(),
      "s+": date.getSeconds()
    }

    //遍历得到obj里面的正则表达式和对应的值
    for (const key in obj) {
      //将得到对象里面 key 的字符串转为正则表达式
      const keyRe = new RegExp(key)
      //判断传入的字符是否存在正则匹配
      //匹配的进行替换
      if (keyRe.test(formatDate)) {
        //内容是单位数的，补充0变为两位数
        const value = (obj[key] + "").padStart(2, "0")
        //替换传入日期格式里面的字符
        formatDate = formatDate.replace(keyRe, value)
      }
    }
    return formatDate
  }

  const result = formatDate(1733231045322, "yyyy-MM-dd hh:mm:ss")
  console.log(result)
  //2024-12-03 21:04:05
</script>
```
