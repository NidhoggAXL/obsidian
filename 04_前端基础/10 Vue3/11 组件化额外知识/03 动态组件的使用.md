# 二、v-if显示不同的组件

```html
<template>
 <div class="app">
  <div class="tabs">
   <template v-for="item in tabs" :key="item">
    <button 
    @click="tabBtn(item)"
    :class="{'active': currentTab === item}"
    >
    {{ item }}
   </button>
   </template>
  </div>
  <!-- 显示是哪一个组件 -->
   <!-- 通过v-if来判断来显示不同的组件 -->
  <div class="content">
   <template v-if="currentTab === 'home'">
	    <Home />
   </template>
   <template v-else-if="currentTab === 'about'">
    <About />
   </template>
   <template v-else-if="currentTab === 'contact'">
    <Category />
   </template>
  </div>
 </div>
</template>

<script>
import Home from './Home.vue';
import About from './About.vue';
import Category from './Category.vue';

export default {
 data() {
  return {
   tabs: ['home', 'about', 'contact'],
   currentTab: 'home',
  }
 },
 methods: {
  // 点击按钮切换组件
  tabBtn(item) {
   this.currentTab = item;
  }
 },
 components: {
  Home,
  About,
  Category
 },
}
</script>

<style scoped>
 .active {
  color:red;
 }
</style>
```
