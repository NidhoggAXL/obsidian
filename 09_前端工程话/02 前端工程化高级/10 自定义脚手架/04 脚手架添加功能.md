# 二、项目创建

```js
// 增加具体的一些功能操作
program
  .command("create <project> [...others]")
  .description("创建一个新项目")
  .action(createProjectAction)
```

```js
// createAction.js
 //  引入Node.js的util模块中的promisify方法
const { promisify } = require("util")
//  引入download-git-repo模块，并使用promisify将其转换为返回Promise的函数
const download = promisify(require("download-git-repo")) 

async function createProjectAction(project) {
  try {
    // 从编写的项目模板中clone下来项目
    //  使用download函数下载项目，支持直接从GitHub地址克隆 参数说明： 
    // 1. 第一个参数是下载源，使用direct:前缀指定直接下载，后面跟着GitHub地址和#mian分支/标签 
    // 2. 第二个参数project指定下载的目标目录 
    // 3. 第三个参数是一个配置对象，设置clone为true表示使用git clone方式下载
    await download("direct:github地址#mian", project, { clone: true }) 
  } catch (error) {
    console.log("github连接失败, 请稍后从事")
  }
}

module.exports = {
  createProjectAction
}
```
