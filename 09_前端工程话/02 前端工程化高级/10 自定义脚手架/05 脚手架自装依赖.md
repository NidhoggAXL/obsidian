const { spawn } = require("child_process") //  从child_process模块中解构出spawn方法

/**
概述：该函数用于执行系统命令，通过创建子进程来运行指定的命令，并处理命令的输出和执行结果。
参数：可变参数，用于传递要执行的命令及其参数。
返回值：返回一个Promise对象，根据命令执行结果进行resolve或reject。执行成功时resolve，执行失败时reject并返回退出码。
 */
function execCommand(...args) {
  return new Promise((resolve, reject) => {
    // 1.开启子进程执行命令
    const childProcess = spawn(...args)

    //  使用子进程的标准输出流(stdout)连接到当前进程的标准输出流(stdout)
    childProcess.stout.pipe(process.stdout)
    //  使用子进程的标准错误流(stderr)连接到当前进程的标准错误流(stderr) 这样可以将子进程的输出和错误信息直接显示在当前进程中
    childProcess.stout.pipe(process.stderr) 

    // 3.监听子进程的执行结束，关闭掉
    childProcess.on("close", code => {
      if (code === 0) {
        resolve()
      } else {
        reject(code)
      }
    })
  })
}

module.exports = execCommand
