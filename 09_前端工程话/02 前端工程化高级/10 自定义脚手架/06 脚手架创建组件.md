使用脚手架创建组件的时候，首先就是要使用一个组件的模板，而这个模板的后缀名是 ejs 的所以需要下载 ejs 来解析这个模板，以及 vsCoder 的插件来提供更好的显示

```shell
npm install ejs
```

模板 component.vue.ejs

```vue.ejs 
<script setup>
  import { ref } from 'vue'
  const message = ref('Hello World')
</script>

<template>
  <div class="<%= name %>">
    <h2><%= name %>: {{ message}}</h2>
  </div>
</template>

<style lang="less" scoped>
.<%= name %> {
  color: red;
}
</style>
```

功能模块:

```js
const { program } = require("commander")
const compileEjs = require("../utils/compileEjs")
const writeCpnFile = require("../utils/write-file")

async function addCompnentAction(cpnname) {
  // 1.创建一个组件:编写组件的模板，根据内容给模板中填充数据
  const result = await compileEjs("component.vue.ejs", {
    name: cpnname
  })

  // 2.将模板写入到文件中
  const dest = program.opts().dest || "src/components"
  await writeCpnFile(`${dest}/${cpnname}.vue`, result)
  console.log("组件创建成功", cpnname + ".vue")
}

module.exports = {
  addCompnentAction
}
```

解析 ejs 模块：

```js
const path = require("path")
const ejs = require('ejs')

/**
概述：该函数用于编译 EJS 模板文件，将数据与模板结合生成最终的 HTML 内容。
参数：
  tempName: 字符串类型，表示模板文件的名称
  data: 对象类型，包含要注入模板的数据
返回值：返回一个 Promise 对象，解析后为编译后的 HTML 字符串，如果编译失败则被拒绝并返回错误信息
 */
function compileEjs(tempName, data) {
  return new Promise((resolve, reject) => {
    const tempPath = `../component/${tempName}`
    //  使用path.resolve方法将相对路径转换为绝对路径 __dirname 表示当前模块的目录名 tempPath 是传入的相对路径参数
    const absolutePath = path.resolve(__dirname, tempPath) 

    //  使用ejs.renderFile方法渲染模板文件 absolutePath: 模板文件的绝对路径 data: 传递给模板的数据对象 回调函数: 处理渲染结果
    ejs.renderFile(absolutePath, data, (err, result) => { 
      if (err) {
        console.log("模板编译失败", err)
        reject(err) //  通过reject将错误传递给Promise的调用者
      }
      resolve(result) //  如果渲染成功，通过resolve将结果传递给Promise的调用者
    })
  })
}

module.exports = compileEjs
```

写入文件模块：

```js
const fs = require("fs")

function writeCpnFile(path, content) {
  return fs.promises.writeFile(path, content)
}

module.exports = writeCpnFile
```
