# 一、接口继承

接口和类一样是可以进行继承的，也是使用extends关键字： 并且我们会发现，接口是支持多继承的（类不支持多继承）

```ts
interface IPerson {
  name: string,
  age: number,
  say: () => void
}

interface IDog {
  height: number,
  run: () => void
}

// 继承
interface IAnimal extends IPerson, IDog {
  color: string
}

const animal: IAnimal = {
  name: '小黑',
  age: 1,
  height: 1.2,
  color: 'black',
  say() {
    console.log(this.name, " say")
  },
  run() {
    console.log(this.name, " run")
  }
}
console.log(animal)
/*
{
  name: '小黑',
  age: 1,
  height: 1.2,
  color: 'black',
  say: [Function: say],
  run: [Function: run]
}
*/
```

# 二、接口的实现

接口定义后，也是可以被类实现的： 

- 如果被一个类实现，那么在之后需要传入接口的地方，都可以将这个类传入； 
- 这就是**面向接口开发**；

```ts
interface IPerson {
  name: string;
  age: number;
  say: () => void;
}

interface IDog {
  height: number;
  run: () => void;
}

//面向接口实现(被类实现)
class IAnimal implements IPerson, IDog {
  constructor(public name: string, public age: number, public height: number) {
    this.name = name;
    this.age = age;
    this.height = height;
  }
  say() {
    console.log("say");
  }
  run() {
    console.log("run");
  }
}

function animal(classType: IAnimal) {
  classType.say();//say
  classType.run();//run
}

const p = new IAnimal("张三", 18, 1.88);
animal(p);

console.log(p.name)// 张三
console.log(p.age)// 18
console.log(p.height)// 1.88
console.log(p.say)// [Function: say]
console.log(p.run)// [Function: run]

p.say();//say
p.run();//run

export {};
```
