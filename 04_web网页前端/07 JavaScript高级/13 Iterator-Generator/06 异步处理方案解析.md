# 二、Generator方案
但是上面的代码其实看起来也是阅读性比较差的，有没有办法可以继续来对上面的代码进行优化呢？

```js
<script>
  function requesDate(url) {
    return new Promise((reslove, reject) => {
      setTimeout(() => {
        reslove(url)
      }, 2000);
    })
  }
  
  function* getDate() {
    const res1 = yield requesDate("why")
    const res12 = yield requesDate(res1 + "kobe")
  }
  
  //生成器返回的是一个对象
  const generator = getDate()
  // console.log(generator.next())
  //{value: Promise, done: false}
  generator.next().value.then((res1) => {
    console.log("res:", res1)
    //res: why
    generator.next(res1).value.then((res2) => {
      console.log("res2:", res2)
      //res2: whykobe
    })
  })
</script>
```

# 三、自动执行generator函数
目前我们的写法有两个问题： 

* 第一，我们不能确定到底需要调用几层的Promise关系； 
* 第二，如果还有其他需要这样执行的函数，我们应该如何操作呢？ 

所以，我们可以封装一个工具函数execGenerator自动执行生成器函数：

![gh](https://raw.githubusercontent.com/AXLflechazoPN/Obsidian/main/2024/1732799586000vyy1jz.png)


```js
function exeGenerator(geFn) {
  //获取生成器
  const generator = geFn()
  function exe(res) {
    //获取调用后得到的Promise对象
    const result = generator.next()
    //对Promise对象的 value 进行判断
    //为 flase 进入 if
    if(!result.done) {
      //对 Promise 的 resolve 进行监听
      result.value.then((res) => {
        console.log("res:", res)
        //第一次调用结束，再次回调
        exe(res)
      })
    } else {
      //调用到 Promise 的 value 为 true
      //结束回调
      return
    }
  }
  exe()
}

//把生成器函数作为形参传入
exeGenerator(getDate)
```

# 四、最终方案

```js
<script>
  function requesDate(url) {
    return new Promise((reslove, reject) => {
      setTimeout(() => {
        reslove(url)
      }, 2000);
    })
  }
  
  async function getDate() {
    const res1 = await requesDate("why")
    console.log("res1:", res1)
    const res2 = await requesDate(res1 + "kobe")
    console.log("res2:", res2)
  }
  
  getDate()
</script>
```
