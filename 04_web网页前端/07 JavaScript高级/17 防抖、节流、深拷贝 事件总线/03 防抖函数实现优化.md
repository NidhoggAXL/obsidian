
# 一、防抖函数基本实现(面试)

```js
<script>
  function mydebounce(fn, delay) {
    //记录 setTimeout，并设置初始值为 null
    let timer = null

    const _debounce = function() {
      //如果发生再次触发，将上一次的 setTimeout 清除
      if (timer) clearTimeout(timer)

      //执行 setTimeout 后把这次记录到 timer 中
      timer = setTimeout(() => {
        fn()

        //执行 setTimeout 后将 timer 置为 0
        timer = null
      }, 3000);
    }

    //返回处理好的函数，绑定到 oninput 上面
    return _debounce
  }
</script>

<script>
  const inputEl = document.querySelector("input")
  
  let count = 1
  inputEl.oninput = mydebounce(function() {
    console.log(`发送网络请求${count++}`)
  })
</script>
```

> 输入 mack 后3秒后发送网络请求1
> 在输入 book 后3秒后发送网络请求2

# 二、this绑定优化

```js
<script>
  function mydebounce(fn, delay) {
    //记录 setTimeout，并设置初始值为 null
    let timer = null

    const _debounce = function() {
      //如果发生再次触发，将上一次的 setTimeout 清除
      if (timer) clearTimeout(timer)

      //执行 setTimeout 后把这次记录到 timer 中
      timer = setTimeout(() => {
        //显式绑定 _debounce
        fn.call(this)

        //执行 setTimeout 后将 timer 置为 0
        timer = null
      }, 3000);
    }

    //返回处理好的函数，绑定到 oninput 上面
    return _debounce
  }
</script>

<script>
  const inputEl = document.querySelector("input")
  
  let count = 1
  inputEl.oninput = mydebounce(function() {
    console.log(`发送网络请求${count++}`, this.value)
  })
</script>
```

> 输入 mack 后3秒后发送网络请求1,通过 this.value 得到 mack
> 在输入 book 后3秒后发送网络请求2，通过 this.value 得到 mackbook

# 三、多参数传入优化
```js
<script>
  function mydebounce(fn, delay) {
    //记录 setTimeout，并设置初始值为 null
    let timer = null

    const _debounce = function(...args) {
      //如果发生再次触发，将上一次的 setTimeout 清除
      if (timer) clearTimeout(timer)

      //执行 setTimeout 后把这次记录到 timer 中
      timer = setTimeout(() => {
        //显式绑定 _debounce
        fn.call(this, args)

        //执行 setTimeout 后将 timer 置为 0
        timer = null
      }, 3000);
    }
    
    //返回处理好的函数，绑定到 oninput 上面
    return _debounce
  }
</script>

<script>
  const inputEl = document.querySelector("input")
  
  let count = 1
  //如果传入多个参数
  inputEl.oninput = mydebounce(function(event) {
    console.log(`发送网络请求${count++}`, this.value, event)
  })
</script>
```
