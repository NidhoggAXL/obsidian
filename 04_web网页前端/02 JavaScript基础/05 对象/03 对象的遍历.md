```js
var bar = {
  name: "why",
  age: 18,
  height: 19.8
}

//1.for循环
//获取对象所有的key，这个key是一个对象数组，可迭代
var infokey = Object.keys(bar)
for (var i = 0; i < infokey.length; i++) {
  var key = infokey[i]
  var value = bar[key]
  console.log(key,value)
}

//2.for...in...
for (var key in bar) {
  var value = bar[key]
  console.log(key,value)
}
```
