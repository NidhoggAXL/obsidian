头等函数(first-class function;第一级函数)是指在程序设计语言中，函数被当作头等公民。

* 这意味着，函数可以作为**别的函数的参数、函数的返回值，赋值给变量或存储在数据结构中;**
* 有人主张也应包括支持匿名函数;
* 借鉴**scheme语言**，将函数提升到"第一等公民"(first class)的地位;

通常我们对作为头等公民的编程方式，称之为函数式编程

* Javascript就是符合函数式编程的语言，这个也是**Javascript的一大特点**;


```js
// 函数可以赋值给变量
var foo1 = function() {
  console.log("fool被执行")
}
foo1()//fool被执行

// 函数在变量之间来回传递
var foo2 = foo1
foo2()//fool被执行

// 函数作为另外一个函数的参数
function bar(fn) {
  console.log("fn:" ,fn)
  fn()
}
bar(foo1)
/*
fn: ƒ () {
      console.log("fool被执行")
    }
fool被执行
*/

// 函数作为另外一个函数的返回值
function sayHello() {
  function hi() {
	console.log("hello")
  }
  return hi
}
var fn = sayHello()
fn()
//hello

// 将函数封装到一个数据结构中
var obj = {
  name: "why",
  eatting: function() {
	console.log("eatting")
  }
}
obj.eatting()//eattingh


function bar1() {
  console.log("bar1被执行")
}
function bar2() {
  console.log("bar2被执行")
}
function bar3() {
  console.log("bar3被执行")
}
//事件总线的封装
var bar = [bar1,bar2,bar3]
//[ƒ, ƒ, ƒ]
```
